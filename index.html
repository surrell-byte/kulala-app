  // ----------------------------- VOICE HOOK -----------------------------

  const useVoiceSynthesis = () => {

    const [voices, setVoices] = React.useState([]);

    const [isPlaying, setIsPlaying] = React.useState(false);

    const [isPaused, setIsPaused] = React.useState(false);

    const [currentSentenceIndex, setCurrentSentenceIndex] = React.useState(0);

    const utteranceRef = React.useRef(null);

    const synth = window.speechSynthesis;

    React.useEffect(() => {

      const loadVoices = () => {

        const available = synth.getVoices();

        const preferred = available.filter(v =>

          v.lang.startsWith('en') &&

          (v.name.includes('Google') || v.name.includes('Natural') ||

           v.name.includes('Samantha') || v.name.includes('Karen') ||

           v.name.includes('Daniel') || v.name.includes('Victoria') ||

           v.name.includes('Alex') || v.name.includes('Moira'))

        );

        setVoices(preferred.length ? preferred : available.filter(v => v.lang.startsWith('en')));

      };

      loadVoices();

      if (synth.onvoiceschanged !== undefined) {

        synth.onvoiceschanged = loadVoices;

      }

    }, []);

    const speakSentences = (sentences, options = { rate: 0.85, pitch: 1.0 }) => {

      if (!sentences.length) return;

      if (synth.speaking) synth.cancel();

      let index = 0;

      const speakNext = () => {

        if (index >= sentences.length) {

          setIsPlaying(false);

          setCurrentSentenceIndex(index);

          return;

        }

        setCurrentSentenceIndex(index);

        const utterance = new SpeechSynthesisUtterance(sentences[index]);

        utterance.rate = options.rate;

        utterance.pitch = options.pitch;

        utterance.voice = voices.find(v => v.lang.startsWith('en')) || voices[0];

        utterance.onend = () => {

          index++;

          if (index < sentences.length && !isPaused) speakNext();

          else if (index >= sentences.length) setIsPlaying(false);

        };

        utterance.onerror = () => setIsPlaying(false);

        utteranceRef.current = utterance;

        synth.speak(utterance);

      };

      setIsPlaying(true);

      setIsPaused(false);

      speakNext();

    };

    const pause = () => {

      if (synth.speaking) {

        synth.pause();

        setIsPaused(true);

      }

    };

    const resume = () => {

      if (synth.paused) {

        synth.resume();

        setIsPaused(false);

      }

    };

    const stop = () => {

      synth.cancel();

      setIsPlaying(false);

      setIsPaused(false);

      setCurrentSentenceIndex(0);

    };

    return {

      voices,

      isPlaying,

      isPaused,

      currentSentenceIndex,

      speakSentences,

      pause,

      resume,

      stop,

    };

  };

  const App = () => {

    const [activeTab, setActiveTab] = React.useState('home');

    const [activeFilter, setActiveFilter] = React.useState('Sleep');

    const [favorites, setFavorites] = React.useState(() => {

      const saved = localStorage.getItem('kulala_favorites');

      return saved ? JSON.parse(saved) : [];

    });

    const [user, setUser] = React.useState(() => {

      const saved = localStorage.getItem('kulala_user');

      return saved ? JSON.parse(saved) : { name: 'Dreamer', favoriteThemes: [] };

    });

    const [progress, setProgress] = React.useState(() => {

      const saved = localStorage.getItem('kulala_progress');

      return saved ? JSON.parse(saved) : { streak: 0, lastReadDate: null, totalStoriesRead: 0, categoryReads: {}, nightReads: 0 };

    });

    const [badges, setBadges] = React.useState(() => {

      const saved = localStorage.getItem('kulala_badges');

      return saved ? JSON.parse(saved) : [];

    });

    const [selectedStory, setSelectedStory] = React.useState(null);

    const [modalOpen, setModalOpen] = React.useState(false);

    const [sleepModeActive, setSleepModeActive] = React.useState(false);

    const [sleepTimerId, setSleepTimerId] = React.useState(null);

    const [activeSound, setActiveSound] = React.useState(null);

    const [soundVolume, setSoundVolume] = React.useState(0.3);

    const audioRef = React.useRef(null);

    const voice = useVoiceSynthesis();

    // Load/save to localStorage

    React.useEffect(() => {

      localStorage.setItem('kulala_favorites', JSON.stringify(favorites));

    }, [favorites]);

    React.useEffect(() => {

      localStorage.setItem('kulala_user', JSON.stringify(user));

    }, [user]);

    React.useEffect(() => {

      localStorage.setItem('kulala_progress', JSON.stringify(progress));

    }, [progress]);

    React.useEffect(() => {

      localStorage.setItem('kulala_badges', JSON.stringify(badges));

    }, [badges]);

    // Background sound handling

    React.useEffect(() => {

      if (activeSound && SOUNDS[activeSound]) {

        if (audioRef.current) {

          audioRef.current.pause();

        }

        const audio = new Audio(SOUNDS[activeSound]);

        audio.loop = true;

        audio.volume = soundVolume;

        audio.play();

        audioRef.current = audio;

      } else {

        if (audioRef.current) {

          audioRef.current.pause();

          audioRef.current = null;

        }

      }

      return () => {

        if (audioRef.current) {

          audioRef.current.pause();

        }

      };

    }, [activeSound, soundVolume]);

    // Update progress when a story is completed (listening finished)

    const updateProgress = (story) => {

      const today = new Date().toDateString();

      let newStreak = progress.streak;

      if (progress.lastReadDate === today) {

        // already counted today

      } else if (progress.lastReadDate === new Date(Date.now() - 86400000).toDateString()) {

        newStreak += 1;

      } else {

        newStreak = 1;

      }

      const hour = new Date().getHours();

      const isNight = hour >= 20 || hour < 5;

      const newProgress = {

        ...progress,

        streak: newStreak,

        lastReadDate: today,

        totalStoriesRead: progress.totalStoriesRead + 1,

        categoryReads: {

          ...progress.categoryReads,

          [story.category]: (progress.categoryReads[story.category] || 0) + 1

        },

        nightReads: progress.nightReads + (isNight ? 1 : 0)

      };

      setProgress(newProgress);

      // Check and award badges

      const newBadges = [...badges];

      BADGES.forEach(badge => {

        if (!newBadges.find(b => b.id === badge.id) && badge.condition(newProgress)) {

          newBadges.push({ id: badge.id, name: badge.name, icon: badge.icon, earnedAt: new Date().toISOString() });

        }

      });

      if (newBadges.length !== badges.length) {

        setBadges(newBadges);

        setTimeout(() => alert(`🎉 New badge earned: ${newBadges[newBadges.length-1].name} ${newBadges[newBadges.length-1].icon}`), 500);

      }

    };

    const toggleFavorite = (id) => {

      setFavorites(prev =>

        prev.includes(id) ? prev.filter(i => i !== id) : [...prev, id]

      );

    };

    const openStory = (story) => {

      setSelectedStory(story);

      setModalOpen(true);

    };

    const closeModal = () => {

      voice.stop();

      setModalOpen(false);

      setSelectedStory(null);

      if (sleepTimerId) clearTimeout(sleepTimerId);

      setSleepModeActive(false);

    };

    const startVoiceForStory = (sentences) => {

      voice.speakSentences(sentences, { rate: 0.85, pitch: 1.0 });

      // Mark story as completed when voice ends (simplified: after all sentences)

      const checkCompletion = setInterval(() => {

        if (!voice.isPlaying && voice.currentSentenceIndex >= sentences.length && selectedStory) {

          clearInterval(checkCompletion);

          updateProgress(selectedStory);

          if (sleepModeActive) {

            setTimeout(() => closeModal(), 2000);

          }

        }

      }, 1000);

    };

    const startSleepTimer = (minutes) => {

      if (sleepTimerId) clearTimeout(sleepTimerId);

      if (minutes > 0) {

        const timer = setTimeout(() => {

          voice.stop();

          setSleepModeActive(true);

          setTimeout(() => closeModal(), 2000);

        }, minutes * 60000);

        setSleepTimerId(timer);

      } else if (sleepTimerId) {

        clearTimeout(sleepTimerId);

        setSleepTimerId(null);

      }

    };

    const updateUserName = (newName) => {

      setUser({ ...user, name: newName });

    };

    const updateFavoriteThemes = (themes) => {

      setUser({ ...user, favoriteThemes: themes });

    };

    const logout = () => {

      setUser({ name: 'Dreamer', favoriteThemes: [] });

      setFavorites([]);

    return (

        <StarField />

        <Header onOpenProfile={() => setActiveTab('profile')} />

          {activeTab === 'home' && (

                onVolumeChange={setSoundVolume}

              />

              <CategoryPills activeFilter={activeFilter} onFilterChange={setActiveFilter} />

              <div className="px-4 stories-section">

                <div className="flex justify-between items-center mb-3">

                  <h3 className="text-white font-semibold text-lg">

                    {activeFilter === 'New' ? '✨ New Stories' : 'Featured Stories'}

                  <button className="text-yellow-400 text-xs">See all →</button>

          {activeTab === 'favorites' && (

            <div className="pt-2">

              <div className="px-4 mb-3">

                <h2 className="text-white text-xl font-bold">❤️ Your Favorites</h2>

                <p className="text-indigo-200 text-sm">Stories you've saved for cozy nights</p>

              </div>

              {favoriteStories.length === 0 ? (

                <div className="text-center py-12 px-4">

                  <div className="text-5xl mb-3">❤️</div>

                  <p className="text-white/60">No favorites yet. Tap the heart on any story to save it here.</p>

                </div>

              ) : (

                <StoryGrid

                  stories={favoriteStories}

                  onOpen={openStory}

                  onToggleFavorite={toggleFavorite}

                  favorites={favorites}

                />

              )}

            </div>

          )}

          {activeTab === 'profile' && (

            <ProfileTab

              user={user}

              favorites={favorites}

              progress={progress}

              badges={badges}

              onUpdateName={updateUserName}

              onUpdateFavoriteThemes={updateFavoriteThemes}

              onLogout={logout}

            />

          )}

        </div>

        <BottomNav activeTab={activeTab} onTabChange={setActiveTab} />

        <StoryModal

          story={selectedStory}

          isOpen={modalOpen}

          onClose={closeModal}

          onPlayVoice={startVoiceForStory}

          voiceState={voice}

          onSetTimer={startSleepTimer}

          userName={user.name}

        />

      </div>

    );

  };

  ReactDOM.createRoot(document.getElementById('root')).render(<App />);
